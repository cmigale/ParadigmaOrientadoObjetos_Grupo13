title recuperarPartida

participant juego
participant partida
participant batalla
participant repositorioPartida 
participant party

autoactivation on 

juego->partida:reanudarPartida()

alt [hayPartida= true]
 partida->party: reanudarParty()
 party->partida:party[]
 partida->repositorioPartida:recuperarProgreso()
 batalla->partida:informacionDeBatalla
 juego<-partida:InciarBatalla()
 else [hayPartida= false]
 juego->partida: nuevaPartida()
 partida->party: invocarPersonajes()
 partida<-party: personajes
 partida->batalla: iniciarBatalla()
 batalla->batalla: agregarEnemigos()
 end
